## Commands

```bash
# Create package structure template
function create_package() {
  local name=$1
  mkdir -p packages/$name/src/__tests__
  cd packages/$name
  npm init -y
}

# Usage:
# create_package "core"
# create_package "adapter-ws"
# create_package "client"
# create_package "react"
# create_package "server"
```
